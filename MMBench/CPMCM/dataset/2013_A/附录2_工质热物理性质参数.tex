\section*{附录2 工质热物理性质参数}

空气、纯燃气、燃气的焓和熵（其中 $T$ 为温度，$f$ 为油气比）

\subsection*{1 空气的焓}
\[
H_{air} = -0.30183674 \times 10^1
+ 0.10489652 \times 10^{-3} T
- 0.23284057 \times 10^{-7} T^2
+ 0.45288431 \times 10^{-11} T^3
- 0.31308477 \times 10^{-15} T^4
+ 0.11341362 \times 10^{-19} T^5
- 0.21298087 \times 10^{-24} T^6
+ 0.16363600 \times 10^{-29} T^7
\]

\subsection*{2 纯燃气的焓}
\[
H_{st} = -0.11152575 \times 10^6
- 0.31020206 \times 10^{-3} T
+ 2.9961197 \times 10^{-7} T^2
- 0.27934788 \times 10^{-11} T^3
+ 0.18746407 \times 10^{-15} T^4
+ 0.73499597 \times 10^{-19} T^5
- 0.15062602 \times 10^{-24} T^6
+ 0.12510984 \times 10^{-29} T^7
\]

\subsection*{3 燃气的焓}
\[
H_{gas} = \frac{H_{air} + f H_{st}}{1+f}
\]
其中 $f$ 为油气比。

---

\subsection*{4 空气的熵}
\[
S_{air} = \left(0.10489652 \times 10^{-3}\right)\ln \left(\frac{T}{10}\right)
+ 0.80558643 \times 10^{-4}
+ \left(-465.6811 T + 0.6793 T^2 - 4.1745 \times 10^{-7} T^3
+ 1.4177 \times 10^{-11} T^4 - 2.5558 \times 10^{-15} T^5
+ 2.2909 \times 10^{-19} T^6 \right)\times 10^{-3}
\]

\subsection*{5 纯燃气的熵}
\[
S_{st} = \left(-0.31020206 \times 10^{-3}\right) \ln \left(\frac{T}{10}\right)
-0.17800633 \times 10^{-4}
+ 10^{-3}\left(5.9922 \times 10^{-3} T - 4.1902 T^2 + 0.0025 T^3
- 9.1874 \times 10^{-7} T^4 + 1.8075 \times 10^{-11} T^5
- 1.4596 \times 10^{-15} T^6 \right)
\]

\subsection*{6 燃气的熵}
\[
S_{gas} = \frac{S_{air} + f S_{st}}{1+f}
\]
其中 $f$ 为油气比。
